(NAST
  (VALUE x_sq
    (Call :multiply
    (InputRef [:x])
    (InputRef [:x])
  )
  )
  (VALUE y_sq
    (Call :multiply
    (InputRef [:y])
    (InputRef [:y])
  )
  )
  (VALUE distance_sq
    (Call :add
    (Ref x_sq)
    (Ref y_sq)
  )
  )
  (VALUE distance
    (Call :power
    (Ref distance_sq)
    (Const 0.5)
  )
  )
)
