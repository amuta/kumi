(SNAST
  (TRAIT y_positive
    (Call :core.gt
      (InputRef [:y]) :: [] -> integer
      (Const 0) :: [] -> integer
    ) :: [] -> boolean
  ) :: [] -> boolean
  (TRAIT x_positive
    (Call :core.gt
      (InputRef [:x]) :: [] -> integer
      (Const 0) :: [] -> integer
    ) :: [] -> boolean
  ) :: [] -> boolean
  (VALUE status
    (Call :__select__
      (Call :core.and
        (Ref y_positive) :: [] -> boolean
        (Ref x_positive) :: [] -> boolean
      ) :: [] -> boolean
      (Const "both positive") :: [] -> string
      (Call :__select__
        (Ref x_positive) :: [] -> boolean
        (Const "x positive") :: [] -> string
        (Call :__select__
          (Ref y_positive) :: [] -> boolean
          (Const "y positive") :: [] -> string
          (Const "neither positive") :: [] -> string
        ) :: [] -> string
      ) :: [] -> string
    ) :: [] -> string
  ) :: [] -> string
)
