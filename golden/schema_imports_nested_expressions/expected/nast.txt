  (VALUE interest
    (ImportCall :annual_interest
    (principal (Call :multiply
      (InputRef [:amount])

    ))
    (rate (InputRef [:rate]))
  )
  )

  )
  )

    (Ref total_over_periods)
    (Const 2)
  )
  )
